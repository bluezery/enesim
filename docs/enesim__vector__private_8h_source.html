00082 /* p0 the initial point
00083  * vx the x direction
00084  * vy the y direction
00085  */
00086 static inline void enesim_f16p16_line_f16p16_direction_from(Enesim_F16p16_Line *l,
00087         Enesim_F16p16_Point *p0, Eina_F16p16 vx, Eina_F16p16 vy)
00088 {
00089     l->a = vy;
00090     l->b = -vx;
00091     l->c = eina_f16p16_mul(vx, p0->y) - eina_f16p16_mul(vy, p0->x);
00092 }
00093 
00094 static inline void enesim_line_direction_from(Enesim_Line *l,
00095         Enesim_Point *p0, double vx, double vy)
00096 {
00097     l->a = vy;
00098     l->b = -vx;
00099     l->c = (vx * p0->y) - (vy * p0->x);
00100 }
